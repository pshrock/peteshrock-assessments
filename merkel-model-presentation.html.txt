        document.querySelectorAll('.option').forEach(option => {
            option.addEventListener('click', function() {
                const radio = this.querySelector('input[type="radio"]');
                if (radio) {
                    radio.checked = true;
                    this.parentElement.querySelectorAll('.option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    this.classList.add('selected');
                }
            });
        });

        document.getElementById('merkelAssessment').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const formData = new FormData(this);
            const responses = {};
            
            for (let [key, value] of formData.entries()) {
                responses[key] = value;
            }
            
            const scores = calculateScores(responses);
            displayResults(scores, responses);
        });

        function calculateScores(responses) {
            const functionScore = parseInt(responses.function_1 || 0);
            const optimizationScore = parseInt(responses.optimization_1 || 0);
            const resilienceScore = parseInt(responses.resilience_1 || 0);
            const momentumScore = parseInt(responses.momentum_1 || 0);
            const strategyScore = parseInt(responses.strategy_1 || 0);
            
            const totalScore = functionScore + optimizationScore + resilienceScore + momentumScore + strategyScore;
            const percentageScore = Math.round((totalScore / 25) * 100);
            
            return {
                function_score: functionScore,
                optimization_score: optimizationScore,
                resilience_score: resilienceScore,
                momentum_score: momentumScore,
                strategy_score: strategyScore,
                total_score: totalScore,
                percentage_score: percentageScore
            };
        }

        function displayResults(scores, responses) {
            const assessmentForm = document.getElementById('assessment-form');
            const resultsSection = document.getElementById('results-section');
            const totalScoreElement = document.getElementById('total-score');
            const interpretationElement = document.getElementById('interpretation');
            const recommendationsElement = document.getElementById('recommendations');
            
            assessmentForm.style.display = 'none';
            resultsSection.style.display = 'block';
            
            totalScoreElement.textContent = scores.percentage_score + '%';
            
            let interpretation = '';
            let recommendations = '';
            
            if (scores.percentage_score >= 90) {
                interpretation = `
                    <h3>Elite Performance Level</h3>
                    <p>Your assessment indicates exceptional performance across all FORM™ dimensions. You demonstrate elite-level functioning under pressure, systematic optimization, rapid resilience recovery, strong momentum building, and strategic decision-making capabilities.</p>
                    <p><strong>Key Strengths:</strong> Pressure performance, systematic improvement, rapid recovery, momentum building, strategic thinking</p>
                `;
                recommendations = `
                    <h3>Optimization Recommendations</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Focus on micro-optimizations and advanced performance techniques</li>
                        <li>Consider mentoring others to reinforce your own systematic approach</li>
                        <li>Explore cutting-edge performance science applications</li>
                        <li>Develop contingency strategies for unprecedented challenges</li>
                    </ul>
                `;
            } else if (scores.percentage_score >= 75) {
                interpretation = `
                    <h3>High Performance Level</h3>
                    <p>You demonstrate strong performance capabilities with consistent results across most FORM™ dimensions. There are specific areas where targeted development could elevate you to elite performance levels.</p>
                    <p><strong>Key Strengths:</strong> Generally strong across multiple dimensions with room for specific optimization</p>
                `;
                recommendations = `
                    <h3>Development Recommendations</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Identify and focus on your lowest-scoring FORM™ dimension</li>
                        <li>Implement systematic tracking of performance metrics</li>
                        <li>Develop specific protocols for pressure situations</li>
                        <li>Create structured recovery and momentum-building routines</li>
                    </ul>
                `;
            } else if (scores.percentage_score >= 60) {
                interpretation = `
                    <h3>Developing Performance Level</h3>
                    <p>Your assessment shows solid foundational performance with significant opportunities for improvement across the FORM™ framework. Systematic development in key areas will yield substantial performance gains.</p>
                    <p><strong>Focus Areas:</strong> Building consistency and developing systematic approaches to performance optimization</p>
                `;
                recommendations = `
                    <h3>Development Priorities</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Establish daily performance tracking and reflection protocols</li>
                        <li>Develop pressure training and stress inoculation practices</li>
                        <li>Create systematic approaches to skill development</li>
                        <li>Build resilience through controlled challenge exposure</li>
                    </ul>
                `;
            } else {
                interpretation = `
                    <h3>Foundation Building Level</h3>
                    <p>Your assessment indicates the need for fundamental development across the FORM™ framework. This represents an excellent opportunity to build systematic performance capabilities from the ground up.</p>
                    <p><strong>Priority:</strong> Establishing core performance systems and building foundational resilience</p>
                `;
                recommendations = `
                    <h3>Foundation Building Plan</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Begin with basic performance tracking and self-awareness development</li>
                        <li>Focus on building stress tolerance through gradual exposure</li>
                        <li>Establish consistent training/work routines</li>
                        <li>Develop fundamental resilience and recovery practices</li>
                    </ul>
                `;
            }
            
            interpretationElement.innerHTML = interpretation;
            recommendationsElement.innerHTML = recommendations;
            
            resultsSection.scrollIntoView({ behavior: 'smooth' });
        }
