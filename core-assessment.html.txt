    <script>
        document.querySelectorAll('.option').forEach(option => {
            option.addEventListener('click', function() {
                const radio = this.querySelector('input[type="radio"]');
                if (radio) {
                    radio.checked = true;
                    this.parentElement.querySelectorAll('.option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    this.classList.add('selected');
                }
            });
        });

        document.getElementById('coreAssessment').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const formData = new FormData(this);
            const responses = {};
            
            for (let [key, value] of formData.entries()) {
                responses[key] = value;
            }
            
            const scores = calculateScores(responses);
            displayResults(scores, responses);
        });

        function calculateScores(responses) {
            const characterScore = parseInt(responses.character_1 || 0);
            const ownershipScore = parseInt(responses.ownership_1 || 0);
            const resilienceScore = parseInt(responses.resilience_1 || 0);
            const excellenceScore = parseInt(responses.excellence_1 || 0);
            
            const totalScore = characterScore + ownershipScore + resilienceScore + excellenceScore;
            const percentageScore = Math.round((totalScore / 20) * 100);
            
            return {
                character_score: characterScore,
                ownership_score: ownershipScore,
                resilience_score: resilienceScore,
                excellence_score: excellenceScore,
                total_score: totalScore,
                percentage_score: percentageScore
            };
        }

        function displayResults(scores, responses) {
            const assessmentForm = document.getElementById('assessment-form');
            const resultsSection = document.getElementById('results-section');
            const totalScoreElement = document.getElementById('total-score');
            const dimensionScoresElement = document.getElementById('dimension-scores');
            const interpretationElement = document.getElementById('interpretation');
            const recommendationsElement = document.getElementById('recommendations');
            
            assessmentForm.style.display = 'none';
            resultsSection.style.display = 'block';
            
            totalScoreElement.textContent = scores.percentage_score + '%';
            
            dimensionScoresElement.innerHTML = `
                <div class="dimension-card">
                    <h4>Character</h4>
                    <div class="score">${scores.character_score}/5</div>
                </div>
                <div class="dimension-card">
                    <h4>Ownership</h4>
                    <div class="score">${scores.ownership_score}/5</div>
                </div>
                <div class="dimension-card">
                    <h4>Resilience</h4>
                    <div class="score">${scores.resilience_score}/5</div>
                </div>
                <div class="dimension-card">
                    <h4>Excellence</h4>
                    <div class="score">${scores.excellence_score}/5</div>
                </div>
            `;
            
            let interpretation = '';
            let recommendations = '';
            
            if (scores.percentage_score >= 85) {
                interpretation = `
                    <h3>Strong Character Foundation</h3>
                    <p>Your COREsuccess™ assessment indicates excellent character development across all dimensions. You demonstrate strong moral foundations, personal responsibility, resilience, and commitment to excellence.</p>
                    <p><strong>Key Strengths:</strong> Integrity, accountability, perseverance, high standards</p>
                `;
                recommendations = `
                    <h3>Character Leadership Development</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Take on leadership roles that allow you to model strong character</li>
                        <li>Mentor others in character development and decision-making</li>
                        <li>Set ambitious goals that challenge you to maintain your standards</li>
                        <li>Continue to refine your ability to make difficult ethical decisions</li>
                    </ul>
                `;
            } else if (scores.percentage_score >= 70) {
                interpretation = `
                    <h3>Developing Character Foundation</h3>
                    <p>You show solid character development with opportunities for continued growth. You have strong foundations in most areas with specific dimensions ready for focused development.</p>
                    <p><strong>Key Strengths:</strong> Generally positive character traits with room for refinement</p>
                `;
                recommendations = `
                    <h3>Character Strengthening Plan</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Focus on your lowest-scoring CORE dimension for targeted improvement</li>
                        <li>Practice making good choices even when it's difficult or inconvenient</li>
                        <li>Find mentors or role models who demonstrate strong character</li>
                        <li>Join activities or groups that challenge you to grow personally</li>
                    </ul>
                `;
            } else {
                interpretation = `
                    <h3>Building Character Foundation</h3>
                    <p>Your assessment shows significant opportunities for character development across the COREsuccess™ framework. This is an excellent time to focus on building strong character habits and decision-making skills.</p>
                    <p><strong>Focus Areas:</strong> Developing consistency in character-based decision making and personal accountability</p>
                `;
                recommendations = `
                    <h3>Character Foundation Building</h3>
                    <ul style="margin: 12px 0; padding-left: 20px;">
                        <li>Start with small daily acts of honesty, kindness, and responsibility</li>
                        <li>Practice taking ownership of your actions and their consequences</li>
                        <li>Set achievable goals and work consistently to accomplish them</li>
                        <li>Seek guidance from trusted adults, mentors, or counselors</li>
                    </ul>
                `;
            }
            
            interpretationElement.innerHTML = interpretation;
            recommendationsElement.innerHTML = recommendations;
            
            resultsSection.scrollIntoView({ behavior: 'smooth' });
        }
